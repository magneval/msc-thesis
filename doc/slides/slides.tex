\begin{frame}
\frametitle{Bevezetõ}
\begin{itemize}
\item Szoftverfejlesztéshez már bevett eszköz manapság a verziókezelõ használata.
\item Ezek elõnyeit szélesebb körben, technikai elõképzettséggel nem rendelkezõ felhasználók is szeretnék élvezni.
\item Erre nyújtanak megoldást a dokumentummenedzsment rendszerek.
\item A kliensek nyílt forráskódú migrációját nehezíti, ha nem tudnak ezekkel a rendszerekkel kommunikálni.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Motiváció}
\begin{figure}[H]
\includegraphics[width=80mm,keepaspectratio]{general-arch-of-doc-mgmt-systems-hu.pdf}
\end{figure}
\begin{itemize}
\item Az egyik legelterjedtebb dokumentummenedzsment rendszer a Sharepoint.
\item Létezik nyílt forráskódú alternatívája, pl. Alfresco.
\item Ehhez létezik OpenOffice.org extension.
\item Írjuk át az OpenOffice.org extensiont, hogy Sharepointtal tudjon kommunikálni!
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Feladat leírása}
\begin{itemize}
\item Sharepoint funkcionalitás felmérése.
\item OpenOffice.org extension készítés elsajátítása.
\item Létezõ extension átírása Sharepointtal való mûködésre.
\item Alfrescohoz és Sharepointhoz lehessen egyidejûleg kapcsolódni.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Megoldott feladatok}
\framesubtitle{Sharepoint funkcionalitás felmérése}
\begin{itemize}
\item Hitelesítés: HTTP Basic, NTLM.
\item Használt protokollok: Vermeer RPC, SOAP.
\item Használati esetek gyûjtése:
\begin{itemize}
\item Workspace létrehozás, törlés.
\item Megnyitás, mentés.
\item Verziók létrehozása, listázása, törlése, visszaállítása.
\item Nem támogatott: felhasználókezelés, linkek, feladatok.
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Megoldott feladatok}
\framesubtitle{Létezõ extension átírása Sharepointtal való mûködésre}
\begin{itemize}
\item Protokoll visszafejtése: Wireshark, vázlatos referencia MSDN-rõl.
\item Megfigyelt rendszer: Microsoft Office 2007, Microsoft Sharepoint 2007.
\item OPAL (OpenOffice Plugin for Alfresco) Java kódjának átírása Sharepointhoz.
\item Különálló Sharepoint library.
\item Szerveroldali komponens telepítése nem szükséges.
\end{itemize}

\frametitle{Megoldott feladatok}

\begin{itemize}

\item Problémák: hiányos implementáció.
\end{itemize}

Összefoglalás:
\begin{itemize}
\item A félév során egy -- nyílt forráskódú irodai programcsomagból használható -- egyszerû Sharepoint kliens készült el.
\item Ennek részeként elkészült egy különállú Sharepoint Java kliens könyvtár, mely korábban nem volt elérhetõ.
\item Használatához szerveroldali komponens telepítése nem szükséges.
\item Több platformon (OpenOffice, LibreOffice) fut, operációs rendszerek közötti hordozhatóságát a Java biztosítja.
